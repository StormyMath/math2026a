            \hline
            #2 \\ \hline 
            #3 \\ 
            \noalign{\global\arrayrulewidth=1.5pt}\arrayrulecolor{black}\hline
            \noalign{\global\arrayrulewidth=0.4pt}\arrayrulecolor{black!30}
            #4 \\ \hline
        \end{tabular}
        \arrayrulecolor{black}
    \end{minipage}
}
